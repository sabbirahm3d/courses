                Let $n$ be represented by $m$, such that,
                \begingroup
                \addtolength{\jot}{1em}
                \begin{align*}
                    m \cdot (57 - m) & = 680\\
                    \Rightarrow 0 & = m^2 - 57m + 680\\
                    & = \frac{57 \pm \sqrt{57^2-4\cdot 680}}{2}\\
                    & = \frac{57 \pm 23}{2}\\
                    m & = 40, 17
                \end{align*}
                \endgroup
                $\therefore m = 40$, $n = 17$ \qedhere
            \end{proof}
